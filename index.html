Telegram бот для продаж

Python • Telebot • JSON • Logging

Полная система маркетплейса цифровых товаров с обработкой заказов и промокодами.

  • Каталог товаров с ценами
  • Обработка заказов
  • Управление промокодами
  • Логирование и обработка ошибок
Результат: 100% автоматизация обработки

Полный исходный код

#!/usr/bin/env python3 import telebot from telebot import types import json, os, logging from datetime import datetime logging.basicConfig(level=logging.INFO) logger = logging.getLogger(__name__) BOT_TOKEN = "YOUR_BOT_TOKEN" bot = telebot.TeleBot(BOT_TOKEN) def load_data(f): if os.path.exists(f): with open(f, 'r', encoding='utf-8') as file: return json.load(file) return {} def save_data(f, d): with open(f, 'w', encoding='utf-8') as file: json.dump(d, file, ensure_ascii=False, indent=4) products = {"p1": {"name": "Курс Python", "price": 500}} users = load_data("users.json") @bot.message_handler(commands=['start']) def start(msg): uid = str(msg.chat.id) if uid not in users: users[uid] = {"name": msg.from_user.first_name, "purchases": []} save_data("users.json", users) markup = types.ReplyKeyboardMarkup() markup.add("Магазин", "Заказы") bot.send_message(msg.chat.id, "Добро пожаловать!", reply_markup=markup) @bot.message_handler(func=lambda m: m.text == "Магазин") def shop(msg): markup = types.InlineKeyboardMarkup() for pid, p in products.items(): markup.add(types.InlineKeyboardButton(f"{p['name']} - {p['price']}₽", callback_data=f"buy_{pid}")) bot.send_message(msg.chat.id, "Товары:", reply_markup=markup) @bot.callback_query_handler(func=lambda c: c.data.startswith("buy_")) def buy(call): pid = call.data.split("_")[1] product = products[pid] uid = str(call.from_user.id) users[uid]["purchases"].append({"product": product["name"], "price": product["price"]}) save_data("users.json", users) bot.send_message(call.message.chat.id, f"Куплено: {product['name']}") bot.infinity_polling()
Установка: pip install pytelegrambotapi

Монитор цен маркетплейсов

Python • BeautifulSoup • SQLite

Система мониторинга цен в реальном времени с автоматическими уведомлениями.

  • Парсинг цен с сайтов
  • SQLite база данных
  • Система алертов
  • Статистика изменений
Результат: Экономия 50 часов/месяц

Полный исходный код

#!/usr/bin/env python3 import requests, sqlite3, time, logging from bs4 import BeautifulSoup from datetime import datetime logging.basicConfig(level=logging.INFO) logger = logging.getLogger(__name__) def init_db(): conn = sqlite3.connect('prices.db') c = conn.cursor() c.execute('''CREATE TABLE IF NOT EXISTS prices (id INTEGER PRIMARY KEY, product TEXT, price REAL, timestamp TEXT)''') conn.commit() return conn def parse_prices(url): try: r = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'}, timeout=10) soup = BeautifulSoup(r.content, 'html.parser') products = {} for item in soup.find_all('div', class_='product'): name = item.find('h2').text.strip() price = float(item.find('span', class_='price').text.replace('₽','').strip()) products[name] = price return products except Exception as e: logger.error(f"Ошибка: {e}") return None def monitor(): conn = init_db() while True: prices = parse_prices("https://example.com") if prices: c = conn.cursor() for name, price in prices.items(): c.execute("INSERT INTO prices (product, price, timestamp) VALUES (?, ?, ?)", (name, price, datetime.now().isoformat())) logger.info(f"{name}: {price}₽") conn.commit() time.sleep(300) monitor()
Установка: pip install requests beautifulsoup4
