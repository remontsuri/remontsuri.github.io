Монитор цен

Python • BeautifulSoup • SQLite

Система мониторинга цен с автоматическими уведомлениями.

  • Парсинг цен
  • SQLite база
  • Алерты
  • Статистика
Экономия 50 часов/месяц
import requests, sqlite3, time from bs4 import BeautifulSoup conn = sqlite3.connect('prices.db') c = conn.cursor() c.execute('''CREATE TABLE IF NOT EXISTS prices (id INTEGER PRIMARY KEY, product TEXT, price REAL, timestamp TEXT)''') def parse(url): r = requests.get(url, headers={'User-Agent': 'Mozilla/5.0'}) soup = BeautifulSoup(r.content, 'html.parser') products = {} for item in soup.find_all('div', class_='product'): name = item.find('h2').text.strip() price = float(item.find('span', class_='price').text.replace('₽','')) products[name] = price return products while True: prices = parse("https://example.com") for name, price in prices.items(): c.execute("INSERT INTO prices VALUES (NULL, ?, ?, datetime('now'))", (name, price)) conn.commit() time.sleep(300)

import requests, smtplib, time from email.mime.text import MIMEText API_KEY = "YOUR_STEAM_API_KEY" def get_inventory(user_id): url = f"https://api.steampowered.com/IEconItems_730/GetUserInventory/v1" r = requests.get(url, params={"key": API_KEY, "steamid": user_id}) data = r.json() if data.get('result', {}).get('success'): return data['result'].get('items', []) return [] def send_email(subject, body): msg = MIMEText(body) msg['Subject'] = subject server = smtplib.SMTP('smtp.yandex.ru', 587) server.starttls() server.login("email@example.com", "password") server.send_message(msg) server.quit() while True: items = get_inventory("YOUR_STEAM_ID") send_email("CS2 Bot", f"Найдено {len(items)} предметов") time.sleep(60)

Hackintosh EFI бэкап

Python • Ryzen 7 5700X3D

Скрипт для автоматизации бэкапов EFI.

  • Автоматический бэкап
  • Очистка старых
  • Восстановление
Полностью рабочая macOS на AMD
import os, shutil from datetime import datetime EFI_DIR = r'E:\EFI' BACKUP_DIR = r'E:\EFI_BACKUPS' def backup_efi(): os.makedirs(BACKUP_DIR, exist_ok=True) today = datetime.now().strftime('%Y-%m-%d_%H-%M-%S') dest = os.path.join(BACKUP_DIR, f'EFI_{today}') shutil.copytree(EFI_DIR, dest) size = sum(os.path.getsize(os.path.join(d,f)) for d,_,fs in os.walk(dest) for f in fs) print(f"Бэкап: {dest} ({size/1024/1024:.2f} MB)") def cleanup_old(keep=5): backups = sorted([(p, os.path.getctime(p)) for p in [os.path.join(BACKUP_DIR, d) for d in os.listdir(BACKUP_DIR)]], key=lambda x: x[1], reverse=True) for path, _ in backups[keep:]: shutil.rmtree(path) backup_efi() cleanup_old()

Скрипты автоматизации Windows.

Экономия 20 часов/месяц
function Backup-Files { $src = "$env:USERPROFILE\Documents" $dst = "D:\Backups\backup_$(Get-Date -Format 'yyyy-MM-dd')" Copy-Item -Path $src -Destination $dst -Recurse -Force Write-Host "Бэкап: $dst" -ForegroundColor Green } function Monitor-System { $cpu = (Get-WmiObject win32_processor | Measure-Object -property LoadPercentage -Average).Average $os = Get-WmiObject win32_operatingsystem $mem = [math]::Round(($os.TotalVisibleMemorySize - $os.FreePhysicalMemory) / $os.TotalVisibleMemorySize * 100) Write-Host "CPU: $cpu% | RAM: $mem%" -ForegroundColor Yellow } function Clean-Temp { $size = 0 "$env:TEMP", "$env:SystemRoot\Temp" | % { Get-ChildItem $_ -Recurse -Force -EA SilentlyContinue | % { $size += $_.Length Remove-Item $_.FullName -Force -Recurse -EA SilentlyContinue } } Write-Host "Очищено: $([math]::Round($size/1MB)) MB" } Backup-Files; Monitor-System; Clean-Temp
